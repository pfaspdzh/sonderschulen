/* =========================

   SORT

========================= */

function sortData(data, useDistance = false) {

  return data.sort((a, b) => {

    const aPriv = a.privat === "Nein" ? 0 : 1;

    const bPriv = b.privat === "Nein" ? 0 : 1;


    if (aPriv !== bPriv) return aPriv - bPriv;


    if (useDistance) {

      return (a.distance ?? 999999) - (b.distance ?? 999999);

    }


    return (a.name || "").localeCompare(b.name || "", "de");

  });

}

/* =========================

   DISTANCE

========================= */

function distance(lat1, lon1, lat2, lon2) {

  const R = 6371;


  const dLat = (lat2 - lat1) * Math.PI / 180;

  const dLon = (lon2 - lon1) * Math.PI / 180;


  const a =

    Math.sin(dLat/2)**2 +

    Math.cos(lat1*Math.PI/180) *

    Math.cos(lat2*Math.PI/180) *

    Math.sin(dLon/2)**2;


  return 2 * R * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));

}


/* =========================

   FILTER

========================= */

function getFilteredData(baseData) {

  const typus = document.getElementById("typusFilter").value;

  const sonder = document.getElementById("sonderFilter").value;

  const spezial = document.getElementById("spezialFilter").value;

  const stufen = document.getElementById("stufenFilter").value;

  const privat = document.getElementById("privatFilter").value;

  const therapien = document.getElementById("therapienFilter").value;

  const heim = document.getElementById("heimFilter").value;


  return baseData.filter(row => {

    return (

      (!typus || row.typus == typus) &&

      (!sonder || row.sonderschueler == sonder) &&

      (!spezial || row.speziell?.split(";").map(x => x.trim()).includes(spezial)) &&

      (!stufen || row.stufen?.split(";").map(x => x.trim()).includes(stufen)) &&

      (!privat || row.privat == privat) &&

      (!therapien || row.therapien_vor_ort == therapien) &&

      (!heim || heim === "ALL" || heimMap[heim]?.includes(row.heimangebot))

    );

  });

}


/* =========================

   APPLY FILTER

========================= */

function applyFilter() {

  const id = wohnortSelect.value;

  const w = wohnorte.find(x => String(x.id) === String(id));


  let filtered = getFilteredData(locations);


  const useDistance = !!w;


  if (w) {

    filtered = filtered.map(l => ({

      ...l,

      distance: distance(w.lat, w.lng, l.lat, l.lng)

    }));

  }


  filtered = sortData(filtered, useDistance);


  renderTable(filtered);

}

function getComments(id) {

  return comments.filter(c => c.location_id === id);

}


function promptUserDropdown() {

  return new Promise(resolve => {

    overlay.style = "position:fixed;top:0;left:0;width:100%;height:100%;background:rgba(0,0,0,0.4);display:flex;align-items:center;justify-content:center;";

    select.innerHTML =

      users.map(u => `<option value="${u.id}">${u.name}</option>`).join("");


    const btn = document.createElement("button");

    btn.textContent = "OK";


    btn.onclick = () => {

      resolve(select.value);

    };


    box.append(select, btn);

    overlay.appendChild(box);

    document.body.appendChild(overlay);

  });

}


async function submitComment(locationId) {

  const input = document.getElementById(`input_${locationId}`);

  const text = input.value.trim();

  if (!text) return;


  const userId = await promptUserDropdown();

  const user = users.find(u => u.id === userId);

  if (!user) return;


  const newComment = {

    location_id: locationId,

    name: user.name,

    comment: text

  };


  comments.push(newComment);

  renderTable(locations);


  await client.from("location_comments").insert([newComment]);


  input.value = "";

}
